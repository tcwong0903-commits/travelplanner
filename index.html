        const getDatesInRange = (startDate, endDate) => {

          const dates = [];

          let currentDate = new Date(startDate);

          const end = new Date(endDate);

          while (currentDate <= end) {

            dates.push(currentDate.toISOString().split('T')[0]);

            currentDate.setDate(currentDate.getDate() + 1);

          }

          return dates;

        };


        const resizeImage = (file, callback) => {

          const reader = new FileReader();

          reader.onload = (event) => {

            const img = new Image();

            img.onload = () => {

              const canvas = document.createElement('canvas');

              const MAX_WIDTH = 600;

              let width = img.width;

              let height = img.height;

              if (width > MAX_WIDTH) {

                height = Math.round((height *= MAX_WIDTH / width));

                width = MAX_WIDTH;

              }

              canvas.width = width;

              canvas.height = height;

              const ctx = canvas.getContext('2d');

              ctx.drawImage(img, 0, 0, width, height);

              callback(canvas.toDataURL('image/jpeg', 0.6));

            };

            img.src = event.target.result;

          };

          reader.readAsDataURL(file);

        };

            if (direction === 'up' && index === 0) return;

            if (direction === 'down' && index === dayEvents.length - 1) return;


            const newOrder = [...dayEvents];

            const swapIdx = direction === 'up' ? index - 1 : index + 1;

            

            [newOrder[index], newOrder[swapIdx]] = [newOrder[swapIdx], newOrder[index]];

            

            const updates = newOrder.map((ev, i) => 

              getDocRef('trip_events', ev.id).update({ orderIndex: i })

            );

        const LedgerTab = ({ trip, expenses }) => {

          const [isExpenseModalOpen, setIsExpenseModalOpen] = useState(false);


          const summary = useMemo(() => {

            const balances = {}; 

            let totalHKD = 0;


            expenses.forEach(exp => {

              const expTotalHKD = parseFloat(exp.amount) * parseFloat(exp.rate);

              totalHKD += expTotalHKD;

              

              const parts = exp.participants || [];

              const numParts = parts.length;

              if (numParts === 0) return;


              const perPerson = expTotalHKD / numParts;

              const settled = exp.settled || []; 


              parts.forEach(p => {

                if (p !== exp.payer && !settled.includes(p)) {

                    balances[exp.payer] = (balances[exp.payer] || 0) + perPerson;

                    balances[p] = (balances[p] || 0) - perPerson;

                }

              });

            });


            const debtors = [];

            const creditors = [];


            for (const [person, balance] of Object.entries(balances)) {

              if (balance < -0.01) debtors.push({ person, amount: -balance });

              else if (balance > 0.01) creditors.push({ person, amount: balance });

            }


            const settlements = [];

            let i = 0, j = 0;

            while (i < debtors.length && j < creditors.length) {

              const debtor = debtors[i];

              const creditor = creditors[j];

              const amount = Math.min(debtor.amount, creditor.amount);


              settlements.push({ from: debtor.person, to: creditor.person, amount });


              debtor.amount -= amount;

              creditor.amount -= amount;


              if (debtor.amount < 0.01) i++;

              if (creditor.amount < 0.01) j++;

            }


            return { totalHKD, balances, settlements };

          }, [expenses]);

        function App() {

          const [user, setUser] = useState(null);

          

          const [trips, setTrips] = useState([]);

          const [events, setEvents] = useState([]);

          const [expenses, setExpenses] = useState([]);

          const [notes, setNotes] = useState([]); 

          

          const [currentView, setCurrentView] = useState('home'); 

          const [currentTripId, setCurrentTripId] = useState(null);

          const [tripSubView, setTripSubView] = useState('timeline'); 

          

          const [isTripModalOpen, setIsTripModalOpen] = useState(false);

          const [editingTrip, setEditingTrip] = useState(null);


          useEffect(() => {

            const initAuth = async () => {

              if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

                await auth.signInWithCustomToken(__initial_auth_token);

              } else {

                try {

                    await auth.signInAnonymously();

                } catch(e) {

                    console.log("登入被略過（通常發生在沒有設定 Firebase Config 的本機環境）");

                    setUser({ uid: "local-dummy-user" });

                }

              }

            };

            initAuth();

            

            const unsubscribe = auth.onAuthStateChanged((u) => {

                if(u) setUser(u);

            });

            return () => unsubscribe();

          }, []);


          useEffect(() => {

            if (!user) return;

            

            try {

                const unsubTrips = getColRef('trips').onSnapshot((snap) => setTrips(snap.docs.map(d => ({ id: d.id, ...d.data() }))), console.error);

                const unsubEvents = getColRef('trip_events').onSnapshot((snap) => setEvents(snap.docs.map(d => ({ id: d.id, ...d.data() }))), console.error);

                const unsubExpenses = getColRef('trip_expenses').onSnapshot((snap) => setExpenses(snap.docs.map(d => ({ id: d.id, ...d.data() }))), console.error);

                const unsubNotes = getColRef('trip_notes').onSnapshot((snap) => setNotes(snap.docs.map(d => ({ id: d.id, ...d.data() }))), console.error);


                return () => { unsubTrips(); unsubEvents(); unsubExpenses(); unsubNotes(); };

            } catch(e) {

                console.warn("Firebase 監聽器啟動失敗，請確認已設定環境變數");

            }

          }, [user]);


          const currentTrip = trips.find(t => t.id === currentTripId);

          const tripEvents = events.filter(e => e.tripId === currentTripId);

          const tripExpenses = expenses.filter(e => e.tripId === currentTripId);

          const tripShopping = notes.filter(n => n.tripId === currentTripId && n.type === 'shopping');

          const tripInfo = notes.filter(n => n.tripId === currentTripId && n.type === 'info');


          if (!user) {

            return (

              <div className="min-h-screen bg-[#f3f4f6] flex items-center justify-center p-4">

                <BrutalBox className="text-center font-black text-xl animate-pulse">載入工業核心...</BrutalBox>

              </div>

            );

          }


          return (

            <div className="min-h-screen text-black font-sans pb-28 relative">

              {currentView === 'home' && (

                <HomeView 

                  trips={trips} 

                  onOpenTrip={(id) => { setCurrentTripId(id); setCurrentView('trip'); setTripSubView('timeline'); }} 

                  onAddTrip={() => { setEditingTrip(null); setIsTripModalOpen(true); }}

                />

              )}


              {currentView === 'trip' && currentTrip && (

                <TripView 

                  trip={currentTrip}

                  events={tripEvents}

                  expenses={tripExpenses}

                  shopping={tripShopping}

                  info={tripInfo}

                  subView={tripSubView}

                  setSubView={setTripSubView}

                  onBack={() => { setCurrentView('home'); setCurrentTripId(null); }}

                  onEditTrip={() => { setEditingTrip(currentTrip); setIsTripModalOpen(true); }}

                />

              )}


              {isTripModalOpen && (

                <TripModal 

                  trip={editingTrip} 

                  onClose={() => setIsTripModalOpen(false)} 

                  onSave={async (data) => {

                    try {

                        if (editingTrip) await getDocRef('trips', editingTrip.id).update(data);

                        else await getColRef('trips').add(data);

                        setIsTripModalOpen(false);

                    } catch(e) {

                        console.error(e);

                        alert("無法儲存，請確認資料庫設定");

                    }

                  }}

                  onDelete={async (id) => {

                    try {

                        await getDocRef('trips', id).delete();

                        setIsTripModalOpen(false);

                        if (currentTripId === id) setCurrentView('home');

                    } catch (e) {

                        console.error(e);

                        alert("無法刪除，請確認資料庫設定");

                    }

                  }}

                />

              )}

            </div>

          );

        }


        // 渲染 React 應用

        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>
